# Prisma Schema для Kakebo v2.0

> Новая модель данных с поддержкой гибких категорий, целей, фиксированных трат и бюджетного уравнения

---

## 📋 Полная схема

```prisma
// ============================================================================
// KAKEBO V2.0 — Полная схема данных
// ============================================================================

// Японская система ведения бюджета КаКеБо — Записи о тратах
model KakeboEntry {
  id          String   @id @default(uuid()) @db.Char(36)
  profileId   String   @db.Char(36)
  profile     Profile  @relation(fields: [profileId], references: [id], onDelete: Cascade)

  date        DateTime @map("date") @db.Date  // Конкретный день траты
  year        Int                           // Год (для быстрой агрегации)
  month       Int                           // Месяц (1-12)

  categoryId  String   @db.Char(36)         // Ссылка на категорию
  category    KakeboCategory @relation(fields: [categoryId], references: [id])

  description String   @db.Text             // Что купили?
  amount      Float                         // Сумма траты в условных единицах
  isNecessary Boolean  @default(false)      // Была ли это необходимость?
  
  isAutoGenerated Boolean @default(false)   // Автоматически создана из фиксированной траты?

  createdAt   DateTime @default(now()) @map("created_at")
  updatedAt   DateTime @updatedAt @map("updated_at")

  @@index([profileId, year, month])
  @@index([profileId, date])
  @@index([categoryId])
  @@index([profileId, isAutoGenerated])
  @@map("kakebo_entries")
}

// Категории трат (системные + пользовательские)
model KakeboCategory {
  id          String         @id @default(uuid()) @db.Char(36)
  profileId   String?        @db.Char(36)  // NULL = системная категория
  profile     Profile?      @relation("CategoryOwner", fields: [profileId], references: [id], onDelete: Cascade)

  name        String                        // "Транспорт", "Кафе", "Аренда"
  type        CategoryType  @default(CUSTOM)  // SYSTEM | CUSTOM
  
  parentId    String?        @db.Char(36)   // Родительская категория (иерархия)
  parent      KakeboCategory? @relation("CategoryToParent", fields: [parentId], references: [id])
  children    KakeboCategory[] @relation("CategoryToParent")

  isFixed     Boolean       @default(false)  // Фиксированная трата? (аренда, подписки)
  isEssential Boolean       @default(true)   // Необходимая? (для аналитики)
  
  icon        String?                        // Иконка Lucide (например, "Car", "Home")
  color       String?                        // Цвет для графиков (#339af0)
  order       Int           @default(0)      // Порядок отображения

  entries     KakeboEntry[]
  fixedExpenses KakeboFixedExpense[]

  @@unique([profileId, name])                // Уникальное имя для пользователя
  @@unique([profileId, parentId, name])      // Уникальное имя в рамках родителя
  @@index([profileId, type])
  @@index([parentId])
  @@index([type])
  @@map("kakebo_categories")
}

enum CategoryType {
  SYSTEM   // Предопределённые (нельзя удалить/переименовать)
  CUSTOM   // Пользовательские (полное управление)
}

// Настройки пользователя Kakebo (глобальные)
model KakeboSettings {
  profileId  String  @id @db.Char(36)
  profile    Profile @relation(fields: [profileId], references: [id], onDelete: Cascade)

  monthLimit Float?  @map("month_limit")    // Устарело: используй KakeboMonthlyBudget.income
  income     Float?                         // Доход за месяц (для бюджетного уравнения)
  currency   String  @default("RUB") @db.VarChar(3)  // Валюта (на будущее)

  createdAt  DateTime @default(now()) @map("created_at")
  updatedAt  DateTime @updatedAt @map("updated_at")

  @@map("kakebo_settings")
}

// Рефлексия пользователя за месяц
model KakeboReflection {
  id             String @id @default(uuid()) @db.Char(36)
  profileId      String @db.Char(36)
  profile        Profile @relation(fields: [profileId], references: [id], onDelete: Cascade)

  year           Int
  month          Int

  // Анализ трат
  unnecessarySpent Float? @map("unnecessary_spent")  // Сумма необязательных трат

  // Ответы на вопросы
  moneyAtStart     Float? @map("money_at_start")     // Сколько было в начале месяца
  plannedToSave    Float? @map("planned_to_save")    // Сколько планировали отложить
  actuallySaved    Float? @map("actually_saved")     // Сколько отложили на самом деле
  improvements     String? @db.Text                  // Что улучшим в следующем месяце

  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  @@unique([profileId, year, month])
  @@index([profileId, year, month])
  @@map("kakebo_reflections")
}

// Цель на месяц («Я обещаю себе…»)
model KakeboMonthlyGoal {
  id          String  @id @default(uuid()) @db.Char(36)
  profileId   String  @db.Char(36)
  profile     Profile @relation(fields: [profileId], references: [id], onDelete: Cascade)

  year        Int
  month       Int

  promise     String  @db.Text                // "Я обещаю себе не покупать кофе навынос"
  targetSave  Float   @map("target_save")     // Сколько хочу отложить (у.е.)
  achieved    Boolean @default(false)         // Достигнута ли цель?

  createdAt   DateTime @default(now()) @map("created_at")
  updatedAt   DateTime @updatedAt @map("updated_at")

  @@unique([profileId, year, month])
  @@index([profileId, year, month])
  @@map("kakebo_monthly_goals")
}

// Фиксированные траты (регулярные платежи)
model KakeboFixedExpense {
  id          String  @id @default(uuid()) @db.Char(36)
  profileId   String  @db.Char(36)
  profile     Profile @relation(fields: [profileId], references: [id], onDelete: Cascade)

  categoryId  String  @db.Char(36)
  category    KakeboCategory @relation(fields: [categoryId], references: [id], onDelete: Restrict)

  description String                         // "Аренда квартиры", "Подписка Netflix"
  amount      Float                          // Сумма ежемесячно

  dayOfMonth  Int     @map("day_of_month")   // День списания (1-31)
  isActive    Boolean @default(true)         // Активна ли трата?

  // Дополнительно (опционально)
  startDate   DateTime? @map("start_date")   // С какого месяца активна
  endDate     DateTime? @map("end_date")     // До какого месяца (если временная)
  notes       String?  @db.Text              // Примечания

  createdAt   DateTime @default(now()) @map("created_at")
  updatedAt   DateTime @updatedAt @map("updated_at")

  @@index([profileId, isActive])
  @@index([categoryId])
  @@index([dayOfMonth])
  @@map("kakebo_fixed_expenses")
}

// Бюджет на месяц (для уравнения)
model KakeboMonthlyBudget {
  profileId  String  @id @db.Char(36)
  year       Int     @id
  month      Int     @id

  profile    Profile @relation(fields: [profileId], references: [id], onDelete: Cascade)

  income     Float                         // Доход за месяц
  // fixedExpenses вычисляется как сумма entries.amount где category.isFixed = true
  // savingGoal берётся из KakeboMonthlyGoal.targetSave
  // remaining вычисляется как income - fixedExpenses - savingGoal

  createdAt  DateTime @default(now()) @map("created_at")
  updatedAt  DateTime @updatedAt @map("updated_at")

  @@unique([profileId, year, month])
  @@index([profileId, year, month])
  @@map("kakebo_monthly_budgets")
}

// ============================================================================
// Отношения в Profile (обновить)
// ============================================================================

// model Profile {
//   // ... существующие поля
//   
//   kakeboEntries         KakeboEntry[]
//   kakeboSettings        KakeboSettings?
//   kakeboReflections     KakeboReflection[]
//   kakeboCategories      KakeboCategory[]          // ← ДОБАВИТЬ
//   kakeboMonthlyGoals    KakeboMonthlyGoal[]       // ← ДОБАВИТЬ
//   kakeboFixedExpenses   KakeboFixedExpense[]      // ← ДОБАВИТЬ
//   kakeboMonthlyBudgets  KakeboMonthlyBudget[]     // ← ДОБАВИТЬ
// }

// ============================================================================
// Индексы для оптимизации
// ============================================================================

// Ключевые индексы для частых запросов:
// 1. Записи за месяц: @@index([profileId, year, month])
// 2. Записи по дате: @@index([profileId, date])
// 3. Категории пользователя: @@index([profileId, type])
// 4. Фиксированные траты: @@index([profileId, isActive])
// 5. Рефлексия/Цели/Бюджет: @@unique([profileId, year, month])
```

## 📊 Схематическое представление

```
Profile (1) ──────< KakeboEntry (N)
                  │
                  └──> categoryId (KakeboCategory)
                       │
                       ├── type: SYSTEM (profileId = NULL)
                       └── type: CUSTOM (profileId = Profile.id)
                       
                       ├── parent (self-relation, иерархия)
                       └── children (self-relation)

Profile (1) ──────< KakeboMonthlyGoal (N)
                  │ (1 запись на месяц)
                  └── year + month

Profile (1) ──────< KakeboFixedExpense (N)
                  │
                  └── categoryId → KakeboCategory
                      │
                      └── автоматически создаёт KakeboEntry

Profile (1) ──────< KakeboMonthlyBudget (N)
                  │ (1 запись на месяц)
                  └── year + month + income
```

## 🔧 Миграция из v1.0

### Шаг 1: Изменить тип ID
```prisma
// v1.0
id String @id @default(cuid())

// v2.0
id String @id @default(uuid()) @db.Char(36)
```

### Шаг 2: Добавить поле categoryId
```prisma
// v1.0
category KakeboCategory @default(LIFE)  // enum

// v2.0
categoryId String @db.Char(36)
category   KakeboCategory @relation(fields: [categoryId], references: [id])

// ← Оставить временно для совместимости:
categoryOld KakeboCategory? @map("category")
```

### Шаг 3: Применить изменения БЕЗ потери данных
```bash
# ❌ НЕ ИСПОЛЬЗОВАТЬ: npx prisma db push --force-reset
#    (это УДАЛИТ ВСЕ данные в kakebo таблицах!)

# ✅ ИСПОЛЬЗОВАТЬ: npx prisma db push
#    (это применит изменения БЕЗ удаления данных)
npx prisma db push
```

### Шаг 4: Создать системные категории
```sql
INSERT INTO kakebo_category (id, name, type, is_essential, order, created_at, updated_at)
VALUES
  ('sys-life', 'Жизнь', 'SYSTEM', true, 1, NOW(), NOW()),
  ('sys-culture', 'Культура', 'SYSTEM', true, 2, NOW(), NOW()),
  ('sys-extra', 'Дополнительное', 'SYSTEM', false, 3, NOW(), NOW()),
  ('sys-unexpected', 'Непредвиденное', 'SYSTEM', false, 4, NOW(), NOW());
```

### Шаг 5: Перенести записи
```sql
UPDATE kakebo_entries 
SET category_id = 'sys-life' 
WHERE category_old = 'LIFE';

-- Повторить для всех категорий
```

### Шаг 6: Удалить старое поле
```prisma
// После проверки данных:
// ← УДАЛИТЬ: categoryOld KakeboCategory? @map("category")
// ← УДАЛИТЬ: enum KakeboCategory { ... }
```
